## 联机

- [freefrp](https://freefrp.net)

[common]
server_addr = frp.freefrp.net
server_port = 7000
token = freefrp.net

[tcp]
type = tcp
local_ip = 127.0.0.1
local_port = 25565
remote_port = 35565
use_encryption = false
use_compression = false

type = udp
local_ip = 127.0.0.1
